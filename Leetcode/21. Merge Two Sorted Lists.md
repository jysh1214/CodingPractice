# [21. Merge Two Sorted Lists](https://leetcode.com/problems/merge-two-sorted-lists/)

1. 
```c++
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* list1, ListNode* list2) {
        if (!list1 && !list2) {
            return nullptr;
        }
        
        ListNode* head = nullptr;
        ListNode* ans = nullptr;
        while (list1 && list2) {
            if (list1->val <= list2->val) {
                if (head) {
                    head->next = list1;
                    head = head->next;
                }
                else {
                    head = list1;
                    ans = head;
                }
                list1 = list1->next;
            }
            else {
                if (head) {
                    head->next = list2;
                    head = head->next;
                }
                else {
                    head = list2;
                    ans = head;
                }
                list2 = list2->next;
            }
        }
        
        if (head) head->next = list1 ? list1 : list2;
        else ans = list1 ? list1 : list2;
        
        return ans;
    }
};
```

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* list1, ListNode* list2) {
        if (!list1 && !list2) {
            return nullptr;
        }
        
        ListNode* head = new ListNode(0);
        ListNode* ans = head;
        while (list1 && list2) {
            if (list1->val <= list2->val) {
                head->next = list1;
                head = head->next;
                list1 = list1->next;
            }
            else {
                head->next = list2;
                head = head->next;
                list2 = list2->next;
            }
        }
        
        head->next = list1 ? list1 : list2;
        
        return ans->next;
    }
};
