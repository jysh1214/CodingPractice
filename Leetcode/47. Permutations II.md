# [47. Permutations II](https://leetcode.com/problems/permutations-ii/)

1. WRONG

不一樣的兩個元素才做交換，但是如果交換後，後半部分全都是重複元素則不會到達終止條件。

```c++
namespace {
    void backtracking(vector<int>& nums, 
                      const int start,
                      vector<vector<int>>& ans) {
        if (start >= nums.size()) {
            ans.push_back(nums);
            return;
        }
        
        for (int i = start; i < nums.size(); ++i) {
            // ... AAAAAA -> 重複元素 不會遞迴下去
            if (nums[start] != nums[i]) {
                swap(nums[start], nums[i]);
                backtracking(nums, start + 1, ans);
                swap(nums[start], nums[i]);
            }
        }
    }
}

class Solution {
public:
    vector<vector<int>> permuteUnique(vector<int>& nums) {
        vector<vector<int>> ans;
        backtracking(nums, 0, ans);
        return ans;
    }
};
```

2. WRONG

不一樣做交換，一樣則下一位繼續做，
但其實毫無意義，跟原始版本排列組合程式碼意思一樣，

若 A, B 一樣
```
swap(A, B);
backtracking(...);
swap(A, B);
```
因 swap 沒有意義 同
```
backtracking(...);
```

```c++
namespace {
    void backtracking(vector<int>& nums, 
                      const int start,
                      vector<vector<int>>& ans) {
        if (start >= nums.size()) {
            ans.push_back(nums);
            return;
        }
        
        for (int i = start; i < nums.size(); ++i) {
            if (nums[start] != nums[i]) {
                swap(nums[start], nums[i]);
                backtracking(nums, start + 1, ans);
                swap(nums[start], nums[i]);
            } else {
                backtracking(nums, start + 1, ans);
            }
        }
    }
}

class Solution {
public:

        vector<vector<int>> ans;
        backtracking(nums, 0, ans);
        return ans;
    }
};
```

3. ACCEPT

先將數列做排序，之後不同元素做交換，並在做完整輪之後才換回來

```c++
namespace {
    void backtracking(vector<int>& nums, 
                      const int start,
                      vector<vector<int>>& ans) {
        if (start >= nums.size()) {
            ans.push_back(nums);
            return;
        }
        
        for (int i = start; i < nums.size(); ++i) {
            if (start != i && nums[start] == nums[i])
                continue;
            
            swap(nums[start], nums[i]);
            backtracking(nums, start + 1, ans);
        }
        
        /**
         * 因為事先排序過，做完整輪之後必定為
         * .....ABBBCCCDDDD -> .....BBBCCCDDDA
         * 將 A 移動回開始的地方
         */
        for(int i = nums.size() - 1; i >= start; --i) {
            swap(nums[i], nums[start]);
        }
    }
}

class Solution {
public:
    vector<vector<int>> permuteUnique(vector<int>& nums) {
        vector<vector<int>> ans;
        sort(nums.begin(), nums.end());
        backtracking(nums, 0, ans);
        return ans;
    }
};
```
