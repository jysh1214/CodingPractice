# [2206. Divide Array Into Equal Pairs](https://leetcode.com/problems/divide-array-into-equal-pairs/)

```c++
class Solution {
public:
    bool divideArray(vector<int>& nums) {
        unordered_map<int, int> count;
        for (int n : nums) {
            count[n] = (count[n] == 0) ? 1 : (count[n] - 1);
        }
        
        auto it = count.begin();
        for (; it != count.end(); ++it) {
            if (it->second > 0)
                return false;
        }
        
        return true;
    }
};
```
